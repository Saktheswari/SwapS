# SwapS
import java.util.Scanner; 
public class SwapS {
   
{
 public static void main(String[]args)
{
int x=100 , y=200;
System.out.println("Before Swap");
System.out.println("x="+x);
System.out.println("y="+y);
int temp = x;
x = y;
y = temp;
System.out.println("After Swap");
System.out.println("x="+x);
System.out.println( "y="+y);
}
}
